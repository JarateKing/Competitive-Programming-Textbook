\newcommand{\inputcode}[2]{
\inputminted[
  fontsize=\scriptsize,
  tabsize=2,
  %bgcolor=CodeBGblank
]{#1}{#2}
}

\newcommand{\inputtcolorbox}{

  width=\linewidth,
  colframe=CodeBGclr,
  colback=CodeBGclr,
  boxsep=1mm,

}

\newcounter{TripleSampleGroupCounter}

\newcommand{\inputall}[1]{

\stepcounter{TripleSampleGroupCounter}

\hspace{-2.4cm}\makebox[\paperwidth]{
\TripleLeftBox{\arabic{TripleSampleGroupCounter}}
\inputcpp{#1.cpp}
\end{tcolorbox} \hfill
\TripleMidBox{\arabic{TripleSampleGroupCounter}}
\inputpython{#1.py}
\end{tcolorbox} \hfill
\TripleRightBox{\arabic{TripleSampleGroupCounter}}
\inputjava{#1.java}
\end{tcolorbox}
}
}

\newcounter{IOsampleGroupCounter}

\newcommand{\IOsample}[1]{

\stepcounter{IOsampleGroupCounter}

\texttt{Input:} \hfill \hspace{4mm} \texttt{Output:} \hfill \hspace{1mm}

\vspace{-2mm}
\IOleftBox{\arabic{IOsampleGroupCounter}}
\inputcode{text}{#1.in}
\end{tcolorbox} \hfill
\IOrightBox{\arabic{IOsampleGroupCounter}}
\inputcode{text}{#1.out}
\end{tcolorbox}
}

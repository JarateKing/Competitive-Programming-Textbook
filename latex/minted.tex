\usepackage{minted}
\usepackage{tcolorbox}

\usepackage{xcolor}
\definecolor{CodeBGclr}{rgb}{0.935,0.935,0.935}

\newcommand{\inputcode}[2]{
\inputminted[
  fontsize=\small,
  tabsize=4
]{#1}{#2}
}

\newcommand{\inputtcolorbox}{
\begin{tcolorbox}[
  width=\linewidth,
  colframe=CodeBGclr,
  colback=CodeBGclr,
  boxsep=1mm,
  arc=3mm
]
}

\newcommand{\inputcpp}[1]{
\inputtcolorbox
\inputcode{cpp}{#1}
\end{tcolorbox}
}

\newcommand{\inputpython}[1]{
\inputtcolorbox
\inputcode{python}{#1}
\end{tcolorbox}
}

\newcommand{\inputjava}[1]{
\inputtcolorbox
\inputcode{java}{#1}
\end{tcolorbox}
}
